def high_mean_points(df, n, value):
    high_mean_indices = np.random.choice(df.index, size=n, replace=False)
    
    df.loc[high_mean_indices, 'avg'] = value
    
    df.loc[high_mean_indices, 'stdev'] = value * 0.2

    )

    plt.xlabel('Point')
    plt.legend()
